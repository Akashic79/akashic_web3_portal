Hợp Đồng Thông Minh Định Danh Phi Tập Trung (DID)

Mỗi thực thể tham gia hệ sinh thái Akashic được cấp một định danh duy nhất dựa trên IOPN. Hợp đồng DID chịu trách nhiệm quản lý quyền truy cập, tuân thủ KYC/AML phi tập trung và bảo vệ dữ liệu người dùng qua bằng chứng không tri thức (ZKP).

  • Tương thích hoàn toàn chuẩn W3C DID.
  • Xác thực không lưu trữ dữ liệu tập trung.
  • Phân quyền động cho các ứng dụng dApp bên thứ ba.
AkashicDID.sol (Demo) 
pragma solidity ^0.8.20;

contract AkashicIdentity {
    struct Identity {
        string iopnDidUri;
        bool isVerified;
        uint256 updatedTimestamp;
    }

    mapping(address => Identity) private _identities;

    event DIDRegistered(address indexed user, string didUri);

    function registerDID(string memory _didUri) public {
        _identities[msg.sender] = Identity(_didUri, true, block.timestamp);
        emit DIDRegistered(msg.sender, _didUri);
    }
}
